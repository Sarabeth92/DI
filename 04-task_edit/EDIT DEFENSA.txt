Algo as√≠:

import { useState } from 'react';
import { Students } from './components/Students';
import { initialStudents } from './data/students';
import { Contador } from './components/Contador';
import { Header } from './components/Header';
import { Buscador } from './components/Buscador';

export const App = () => {
  const [alumnos, setAlumnos] = useState(
    initialStudents.map((student) => ({
      ...student,
      status: 'absent', // estado inicial
    }))
  );

  const handleChangeStatus = (id, newStatus) => {
    setAlumnos((prev) =>
      prev.map((student) =>
        student.id === id ? { ...student, status: newStatus } : student
      )
    );
  };

  return (
    <>
      <Header />
      <main className="bg-gray-100 p-2 m-2">
        <Contador students={alumnos} />
        <Buscador />
        <div className="grid grid-cols-1 md:grid-cols-3 gap-3 px-2">
          {alumnos.map((student) => (
            <Students
              key={student.id}
              student={student}
              onChangeStatus={(newStatus) =>
                handleChangeStatus(student.id, newStatus)
              }
            />
          ))}
        </div>
      </main>
    </>
  );
};

const handleAddStudent = () => {
  const name = window.prompt('Nombre del nuevo estudiante:');
  if (!name) return;

  setAlumnos((prev) => {
    const nextId =
      prev.length > 0 ? Math.max(...prev.map((s) => s.id)) + 1 : 1;

    return [
      ...prev,
      {
        id: nextId,
        name,
        image: 'https://via.placeholder.com/64', // o la que uses t√∫

};
